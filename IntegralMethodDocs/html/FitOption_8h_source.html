148 //need to calculate bin width initally
149 if(rebinChoice)
150 {
151 Double_t rebinSize = (Double_t) numBins;
152 Double_t tempWidth;
153
154 //bin data for Bateman histograms
155 for(int i = 0; i < numCycles; i++)
156 {
157 timeFitEndArr[i] = timeRunEndSimulated;
158 tempWidth = timeRunEndSimulated / rebinSize;
159 binWidthArr[i] = tempWidth;
160 binNumArr[i] = rebinSize;
161 timeFitStartArr[i] = 0.0f;
162 rebinSize = rebinSize + rebinBinInc;
163 tempWidth = timeRunEndSimulated / rebinSize;
164 }
165
166 //bin data for integral histograms

168 {
169 tempWidth = binWidthArr[i];
170 rebinSize = binNumArr[i];

172 binEdgesArr[i] = new Double_t[intRebinSize + 2];
173 CreateShiftedBinEdges(rebinSize, tempWidth, binEdgesArr[i]);
174 }
175 }
176 //need to calculate number of bins initally, change time fit based on number bins
177 if(!rebinChoice && inputHistoExecutionType == 1)
178 {
179 Double_t fitEnd = timeRunEndSimulated;
180 Double_t fitStart = 0.0f;
181 Double_t initBinNum = fitEnd / binWidth;
182 Double_t addedBins = 0.0f;

184 //data for Bateman histograms
185 for(int i = 0; i < numCycles; i++)

187 timeFitEndArr[i] = fitEnd;

198 }else if(timeShiftType == 2)

200 fitStart = addedBins * binWidth;

202 }
203
204 //data for integral histograms
205 for(int i = 0; i < numCycles; i++)
206 {
207 binEdgesArr[i] = new Double_t[binNumArr[i] + 2];
208 CreateShiftedBinEdges(binNumArr[i], binWidthArr[i], binEdgesArr[i]);
209 }
210 }
211 //for input histogram
