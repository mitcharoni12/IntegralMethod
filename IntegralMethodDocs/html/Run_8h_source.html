107Run::Run(ElementFit* element)
108{
109 //getting parameters for the program execution type
110 this->fitOptions = element->GetFitOptions();
111 this->numRuns = fitOptions->GetNumRuns();
112 this->element = element;
113 this->elementNameStrs = fitOptions->GetElementNames();
114 this->timeFitEnd = fitOptions->GetTimeLengthArr();
115 this->inputHistoExecutionType = fitOptions->GetInputHistoExecutionType();
116 this->singleElementDataChoice = fitOptions->GetSingleElementDataChoice();
117 numElements = fitOptions->GetNumElements();
118
119 //dynamical allocation for arrays containing events of general
120 runsXAxis = new Double_t [numRuns];
121 for(int i = 0; i < numRuns; i++)
122 {
123 runsXAxis[i] = i+1;
124 }
125 zero = new Double_t [numRuns];
126 for(int i = 0; i < numRuns; i++)
127 {
128 zero[i] = 0.0f;
129 }
130 //histos used to store fit values of the different runs
131 totalBatemanMultiRunFitHisto = new TH1D* [numElements];
132 totalIntegralMultiRunFitHisto = new TH1D* [numElements];
133 if(singleElementDataChoice == 2)
134 {
135 singleBatemanMultiRunFitHisto = new TH1D* [numElements];
136 singleIntegralMultiRunFitHisto = new TH1D* [numElements];
137 }
138 //used to store fit values for the different runs
139 batemanFitValues = new ChainRunFitValues(numElements, numRuns);
140 integralFitValues = new ChainRunFitValues(numElements, numRuns);
141 if(singleElementDataChoice == 2)
142 {
143 singleBatemanFitValues = new SingleChainRunFitValues(numElements, numRuns);
144 singleIntegralFitValues = new SingleChainRunFitValues(numElements, numRuns);
145 }
146 //used to plot fit values for different runs
147 totalBatemanGraphs = new TGraphErrors* [numElements];
148 totalIntegralGraphs = new TGraphErrors* [numElements];
149 if(singleElementDataChoice == 2)
150 {
151 singleBatemanGraphs = new TGraphErrors* [numElements];
152 singleIntegralGraphs = new TGraphErrors* [numElements];
153 }
154 //testCan = new TCanvas("testCanvas", "testCanvas", 500, 500);
155}

442void Run::RunSingleBatemanRunsGenOnce(Int_t cycleIndex, Int_t runIndex)
443{
444 SingleElementFitValues* singleTempFitParameters;
445 Double_t tempN0;
446 Double_t tempN0Error;
447 Double_t tempHalfLife;
448 Double_t tempHalfLifeError;
449
450 element->FitSingleBatemanHistos(cycleIndex, 0);
451
452 //get single bateman fit parameters
453 singleTempFitParameters = element->GetSingleBatemanFitValues();
454
455 //move single bateman fit parameters into respective class
456 for(int i = 0; i < numElements; i++)
457 {
458 for(int subIndex = 0; subIndex < i+1; subIndex++)
459 {
460 tempN0 = singleTempFitParameters->GetAnN0(i, subIndex);
461 tempN0Error = singleTempFitParameters->GetAnN0Error(i, subIndex);
462 tempHalfLife = singleTempFitParameters->GetAnHalfLife(i, subIndex);
463 tempHalfLifeError = singleTempFitParameters->GetAnHalfLifeError(i, subIndex);
464
465 singleBatemanFitValues->SetAnN0(0, i, subIndex, tempN0);
466 singleBatemanFitValues->SetAnN0Error(0, i, subIndex, tempN0Error);
467 singleBatemanFitValues->SetAnHalfLife(0, i, subIndex, tempHalfLife);
